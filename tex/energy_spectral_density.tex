    \begin{axis}[

        xlabel = {$\omega$},
        ylabel = {$|F(j\omega)|^2$},
        xmin = -7, xmax = 7.2,
        ymin = 0, ymax = 1.2,
        xtick = {0.5, 0.7},
        xticklabels = {,},
